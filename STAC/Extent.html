def from_hash(hash)
  transformed = hash.transform_keys(&:to_sym)
  transformed[:spatial] = Spatial.from_hash(transformed.fetch(:spatial))
  transformed[:temporal] = Temporal.from_hash(transformed.fetch(:temporal))
  new(**transformed)
end
