def all_items
  # The last `.lazy` is not necessary with Ruby 3.1.
  # But with Ruby 3.0, it is necessary because Enumerator::Lazy#chain returns Enumerator::Chain
  # and RBS type check fails.
  items.chain(children.flat_map(&:items)).lazy
end
